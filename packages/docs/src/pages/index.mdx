1. ## Create your theme

  ```javascript
  export default {
    colors: {
      text: '#000',
      background: '#fff',
      primary: 'tomato',
    },
    fonts: {
      body: 'system-ui, sans-serif',
      heading: '"Avenir Next", sans-serif',
    },
  }
  ```

2. ## Style your UI

  ```jsx
  /** @jsx jsx */
  import { jsx } from 'theme-ui'
  export default props => (
    <h1
      sx={{
        color: 'primary',
        fontFamily: 'heading',
      }}>
      Hello
    </h1>
  )
  ```
